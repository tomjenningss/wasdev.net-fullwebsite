{
“params”:[
{
“value”:”hello”,
“type”:”java.lang.String”
},
{
“value”:{
“hello”:”world”
},
“type”:{
“className”:”java.util.HashMap”,
“simpleKey”:true,
“entries”:[
{
“key”:”hello”,
“keyType”:”java.lang.String”,
“value”:”java.lang.String”
}
]
}
}
],
“signature”:[
“java.lang.String”,
“java.util.Map”
]
