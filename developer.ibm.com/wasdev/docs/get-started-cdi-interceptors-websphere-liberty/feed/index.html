@PerformanceLogger
@Singleton
@WebService(serviceName = “Service_v01”,
endpointInterface = “Service”,
targetNamespace = “http://services/”)
@HandlerChain(file = “/handlers.xml”)
@Addressing
public class ServieImpl implements Service{
/**
*method to implement
*/
}
//CDI interceptor
@Inherited
@InterceptorBinding
@Target({ ElementType.METHOD, ElementType.TYPE })
@Retention(RetentionPolicy.RUNTIME)
public @interface PerformanceLogger {
}
//interceptor activation
